    LOCAL PROCEDURE GetDateFilter@1();
    VAR
      VATEntry@1000 : Record 254;
      VATEntry1@1001 : Record 254;
    BEGIN
      IF STRLEN(VATEntry1.GETFILTER("Posting Date")) > 0 THEN BEGIN
        IF STRPOS(VATEntry1.GETFILTER("Posting Date"), '..') = 0 THEN BEGIN
          DateFrom := FORMAT(VATEntry1.GETFILTER("Posting Date"));
          DateTo := FORMAT(VATEntry1.GETFILTER("Posting Date"));
          IF STRLEN(DateTo) > 0 THEN
            DateTo := ' to ' + DateTo;
        END
        ELSE IF STRPOS(VATEntry1.GETFILTER("Posting Date"), '..') = 1 THEN BEGIN
          VATEntry.RESET;
          VATEntry.SETCURRENTKEY("Posting Date");
          VATEntry.SETASCENDING("Posting Date",TRUE);
          IF VATEntry.FINDFIRST THEN BEGIN
            DateFrom := FORMAT((VATEntry."Posting Date"),0,'<Month Text> <Closing><Day>, <Year4>');
          END;
          DateTo := FORMAT((VATEntry1.GETRANGEMAX("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          IF STRLEN(DateTo) > 0 THEN
            DateTo := ' to ' + DateTo;
        END
        ELSE IF (STRPOS(VATEntry1.GETFILTER("Posting Date"), '..') = 9) AND (STRLEN(VATEntry1.GETFILTER("Posting Date")) > 10) THEN BEGIN
          //MESSAGE('910');
          DateFrom := FORMAT((VATEntry1.GETRANGEMIN("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          DateTo := FORMAT((VATEntry1.GETRANGEMAX("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          IF STRLEN(DateTo) > 0 THEN
            DateTo := ' to ' + DateTo;
        END
        ELSE IF STRPOS(VATEntry1.GETFILTER("Posting Date"), '..') = 9 THEN BEGIN
          //MESSAGE('9');
          DateFrom := FORMAT((VATEntry1.GETRANGEMIN("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          VATEntry.RESET;
          VATEntry.SETCURRENTKEY("Posting Date");
          VATEntry.SETASCENDING("Posting Date",TRUE);
          IF VATEntry.FINDLAST THEN BEGIN
            DateTo := FORMAT((VATEntry."Posting Date"),0,'<Month Text> <Closing><Day>, <Year4>');
            IF STRLEN(DateTo) > 0 THEN
              DateTo := ' to ' + DateTo;
          END;
        END;
      END
      ELSE BEGIN
        VATEntry.RESET;
        VATEntry.SETCURRENTKEY("Posting Date");
        VATEntry.SETASCENDING("Posting Date",TRUE);
        IF VATEntry.FINDFIRST THEN
          DateFrom := FORMAT(VATEntry."Posting Date");
        IF VATEntry.FINDLAST THEN
          DateTo := FORMAT(VATEntry."Posting Date");
        IF STRLEN(DateTo) > 0 THEN
          DateTo := ' to ' + DateTo;
      END;
    END;

    BEGIN
    END.
