                                  InternalNoticetoProceed@1001 : Record 50040;

                                  SalesInvoiceLine@1002 : Record 113;
                                  CustLedgerEntry@1004 : Record 21;
                                  AppliedEntry@1005 : Record 21;
                                  QtyDelivered@1000 : Decimal;
                                BEGIN
                                  //COMPANY PICTURE//
                                  CompanyInfo.GET;
                                  CompanyInfo.CALCFIELDS(Picture);
                                  //END//
                                  //

                                  //QTY Delivered//
                                  TotalBilledAmount := 0;
                                  TotalQtyDelivered := 0;
                                  SalesInvoiceHeader.RESET;
                                  SalesInvoiceHeader.SETRANGE("INTP No.", "Internal Notice to Proceed"."No.");
                                  IF SalesInvoiceHeader.FINDSET THEN
                                    REPEAT
                                      SalesInvoiceLine.RESET;
                                      SalesInvoiceLine.SETRANGE("Document No.", SalesInvoiceHeader."No.");
                                      IF SalesInvoiceLine.FINDSET THEN BEGIN
                                        REPEAT
                                          TotalQtyDelivered += SalesInvoiceLine.Quantity;
                                        UNTIL SalesInvoiceLine.NEXT = 0;
                                      END;
                                      SalesInvoiceHeader.CALCFIELDS("Amount Including VAT");
                                      TotalBilledAmount += SalesInvoiceHeader."Amount Including VAT";
                                    UNTIL SalesInvoiceHeader.NEXT = 0;
                                  //END//

                                  //Collected Amount//
                                  CollectedAmount := 0;
                                  SalesInvoiceHeader.RESET;
                                  SalesInvoiceHeader.SETRANGE("INTP No.", "Internal Notice to Proceed"."No.");
                                  IF SalesInvoiceHeader.FINDSET THEN BEGIN
                                    REPEAT
                                      CustLedgerEntry.RESET;
                                      CustLedgerEntry.SETRANGE("Document No.", SalesInvoiceHeader."No.");
                                      IF CustLedgerEntry.FINDSET THEN BEGIN
                                        REPEAT
                                          AppliedEntry.RESET;
                                          AppliedEntry.SETRANGE("Entry No." , CustLedgerEntry."Entry No.");
                                          IF AppliedEntry.FINDSET THEN BEGIN
                                            REPEAT
                                              AppliedEntry.CALCFIELDS(Amount);
                                              CollectedAmount += AppliedEntry.Amount - AppliedEntry."Remaining Amount";
                                            UNTIL AppliedEntry.NEXT = 0;
                                          END;
                                        UNTIL CustLedgerEntry.NEXT = 0;
                                      END;
                                    UNTIL SalesInvoiceHeader.NEXT = 0;
                                  END;
                                  //END//
                                END;
                                 }
