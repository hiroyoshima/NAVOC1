                                      CompanyInformation.Picture.CREATEINSTREAM(InStr);

                                      ItemLedgerEntry.CALCFIELDS("Sales Amount (Actual)");

                                    UNTIL ItemLedgerEntry.NEXT = 0;

                                  //GET SALES
                                  CustomerSales := 0;
                                  Customer.RESET;
                                  Customer.SETRANGE("Date Filter", FromDate, ToDate);
                                  Customer.SETFILTER("Virtual Account No.",'<>%1','');

                                  IF Customer.FINDSET THEN BEGIN
                                    REPEAT
                                      Customer.CALCFIELDS("Sales (LCY)");
                                      Customer.CALCFIELDS("Net Change (LCY)");
                                      CustomerSales += Customer."Sales (LCY)";
                                      CustomerNetChange += Customer."Net Change (LCY)";
                                    UNTIL Customer.NEXT = 0;
                                  END;

                                  //DRAGONPAY REMITTANCE
                                  PaymentGatewaySetup.GET;
                                  IF PaymentGatewaySetup."Service Provider" = PaymentGatewaySetup."Service Provider"::DragonPay THEN BEGIN
                                    PaymentGatewaySetup.TESTFIELD("Dragonpay Account No.");

                                    Customer.RESET;
                                    Customer.SETRANGE("Date Filter", FromDate, ToDate);
                                    Customer.SETRANGE("No.", PaymentGatewaySetup."Dragonpay Account No.");

                                    IF Customer.FINDSET THEN BEGIN
                                      Customer.CALCFIELDS("Net Change (LCY)");
                                      DragonPayRemittance += Customer."Net Change (LCY)";
                                    END;
                                  END;
                                END;
