    PROCEDURE ApplyQuote@3(LocString@1000 : Text[250]) : Text[250];
    VAR
      CharCtr@1001 : Integer;
      ApplyQuote@1002 : Boolean;
    BEGIN
      ApplyQuote := FALSE;
      FOR CharCtr := 1 TO STRLEN(LocString) DO BEGIN
        IF LocString[CharCtr] = ',' THEN BEGIN
          EXIT('"' + LocString + '"');
        END;
      END;

      EXIT (LocString);
