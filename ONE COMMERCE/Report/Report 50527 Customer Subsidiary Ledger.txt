    LOCAL PROCEDURE GetORData@184();
    VAR
      DtldCustLedgEntry1@1001 : Record 379;
      DtldCustLedgEntry2@1000 : Record 379;
    BEGIN

      ORDataEntry.RESET;
      IF Invoice."Entry No." <> 0 THEN BEGIN
        CreateCustLedgEntry := Invoice;
        DtldCustLedgEntry1.SETCURRENTKEY("Cust. Ledger Entry No.");
        DtldCustLedgEntry1.SETRANGE("Cust. Ledger Entry No.", CreateCustLedgEntry."Entry No.");
        DtldCustLedgEntry1.SETRANGE(Unapplied, FALSE);
        IF DtldCustLedgEntry1.FIND('-') THEN
          REPEAT
            IF DtldCustLedgEntry1."Cust. Ledger Entry No." = DtldCustLedgEntry1."Applied Cust. Ledger Entry No." THEN BEGIN
              DtldCustLedgEntry2.INIT;
              DtldCustLedgEntry2.SETCURRENTKEY("Applied Cust. Ledger Entry No.", "Entry Type");
              DtldCustLedgEntry2.SETRANGE("Applied Cust. Ledger Entry No.", DtldCustLedgEntry1."Applied Cust. Ledger Entry No.");
              DtldCustLedgEntry2.SETRANGE("Entry Type", DtldCustLedgEntry2."Entry Type"::Application);
              DtldCustLedgEntry2.SETRANGE(Unapplied, FALSE);
              IF DtldCustLedgEntry2.FIND('-') THEN
                REPEAT
                  IF DtldCustLedgEntry2."Cust. Ledger Entry No." <> DtldCustLedgEntry2."Applied Cust. Ledger Entry No." THEN BEGIN
                    ORDataEntry.SETCURRENTKEY("Entry No.");
                    ORDataEntry.SETRANGE("Entry No.", DtldCustLedgEntry2."Cust. Ledger Entry No.");
                    IF ORDataEntry.FIND('-') THEN
                      ORDataEntry.MARK(TRUE);
                  END;
                UNTIL DtldCustLedgEntry2.NEXT = 0;
            END ELSE BEGIN
              ORDataEntry.SETCURRENTKEY("Entry No.");
              ORDataEntry.SETRANGE("Entry No.", DtldCustLedgEntry1."Applied Cust. Ledger Entry No.");
              IF ORDataEntry.FIND('-') THEN
                ORDataEntry.MARK(TRUE);
            END;
          UNTIL DtldCustLedgEntry1.NEXT = 0;

          ORDataEntry.SETCURRENTKEY("Entry No.");
          ORDataEntry.SETRANGE("Entry No.");

          IF CreateCustLedgEntry."Closed by Entry No." <> 0 THEN BEGIN
            ORDataEntry."Entry No." := CreateCustLedgEntry."Closed by Entry No.";
            ORDataEntry.MARK(TRUE);
          END;

          ORDataEntry.SETCURRENTKEY("Closed by Entry No.");
          ORDataEntry.SETRANGE("Closed by Entry No.", CreateCustLedgEntry."Entry No.");
          IF ORDataEntry.FIND('-') THEN
            REPEAT
              ORDataEntry.MARK(TRUE);
            UNTIL ORDataEntry.NEXT = 0;

      END;

      ORDataEntry.MARKEDONLY(TRUE);

      // IF ORDataEntry.FIND('-') THEN
      //  REPEAT
      //    GLBankName := '';
      //    GLAccountNo := '';
      //    CASE ORDataEntry."Bal. Account Type" OF
      //      ORDataEntry."Bal. Account Type"::"Bank Account":
      //        BEGIN
      //          MESSAGE(ORDataEntry."Bal. Account No.");
      //          IF BankAccount.GET(ORDataEntry."Bal. Account No.") THEN
      //            GLBankName := BankAccount.Name + ' ' + BankAccount."Name 2";
      //            GLAccountNo := BankAccount."No.";

      //            GLBankName := GLAccount.Name;
      //            GLAccountNo := GLAccount."No.";
      //        END;
      //    END;

    END;
