    OnPreReport=BEGIN
                  JobLedgerEntry.RESET;
                  //JobLedgerEntry.COPYFILTER("No.", "Job Ledger Entry"."No.");
                  //JobLedgerEntry.COPYFILTERS("Job Ledger Entry");
                  IF JobLedgerEntry.FINDSET THEN BEGIN
                    REPEAT
                      IF JobLedgerEntry.Quantity < 0 THEN BEGIN
                        IF NegativeRecord <> '' THEN BEGIN
                          NegativeRecord += '&';
                          FilterRecord += '|';
                        END;
                        NegativeRecord += '<>' + FORMAT(JobLedgerEntry."Entry No.");
                        FilterRecord += FORMAT(JobLedgerEntry."Entry No.");
                      END;
                    UNTIL JobLedgerEntry.NEXT = 0;
                  END;

                  JobLedgerEntry2.RESET;
                  //JobLedgerEntry2.COPYFILTER("No.", "Job Ledger Entry"."No.");
                  //JobLedgerEntry2.COPYFILTERS("Job Ledger Entry");
                  JobLedgerEntry2.SETFILTER("Entry No.", FilterRecord);
                  IF JobLedgerEntry2.FINDSET THEN BEGIN
                    REPEAT
                      JobLedgerEntry3.RESET;
                      JobLedgerEntry3.COPYFILTERS("Job Ledger Entry");
                      JobLedgerEntry3.SETFILTER("Job No.", JobLedgerEntry2."Job No.");
                      JobLedgerEntry3.SETFILTER(Quantity, '>0');
                      IF JobLedgerEntry3.FINDSET THEN BEGIN
                        REPEAT
                          IF (JobLedgerEntry3."Job No." = JobLedgerEntry2."Job No.") AND
                            (JobLedgerEntry3."No." = JobLedgerEntry2."No.") AND
                            (JobLedgerEntry3."Unit of Measure Code" = JobLedgerEntry2."Unit of Measure Code") AND
                            (JobLedgerEntry3."Job Task No." = JobLedgerEntry2."Job Task No.") THEN BEGIN
                              IF NegativeRecord <> '' THEN
                                NegativeRecord += '&';
                              NegativeRecord += '<>' + FORMAT(JobLedgerEntry3."Entry No.");
                          END;
                        UNTIL JobLedgerEntry3.NEXT = 0;
                      END;
                    UNTIL JobLedgerEntry2.NEXT = 0;
                  END;
                  //MESSAGE(FORMAT(NegativeRecord));
                END;
