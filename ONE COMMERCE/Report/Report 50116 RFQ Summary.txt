Public Function Rate(ByVal Status as String, ByVal CompletedDate as String, ByVal Timeline as String)

If Status = "Completed" Then 
Dim FormatCompletedDate, FormatTimeline as Date

FormatCompletedDate = CDate(CompletedDate)
FormatTimeline = CDate(Timeline)

If FormatCompletedDate &lt; FormatTimeline Then
	return 5
ElseIf FormatCompletedDate = FormatTimeline Then
	return 4
ElseIf DateDiff("D", FormatCompletedDate, FormatTimeline) = -1 Then
	return 3
ElseIf DateDiff("D", FormatCompletedDate, FormatTimeline) = -2 Then
	return 2
Else
	return 1
End If
End If
