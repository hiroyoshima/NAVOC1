    LOCAL PROCEDURE GetDateFilter@5();
    BEGIN

      IF STRLEN("G/L Entry".GETFILTER("Posting Date")) > 0 THEN BEGIN
        IF STRPOS("G/L Entry".GETFILTER("Posting Date"), '..') = 0 THEN BEGIN
          //MESSAGE('0');
          DateFrom := FORMAT("G/L Entry".GETFILTER("Posting Date"));
          DateTo := FORMAT("G/L Entry".GETFILTER("Posting Date"));
          IF STRLEN(DateTo) > 0 THEN
            DateTo := ' to ' + DateTo;
        END
        ELSE IF STRPOS("G/L Entry".GETFILTER("Posting Date"), '..') = 1 THEN BEGIN
          //MESSAGE('1');
          GLEntries.RESET;
          GLEntries.SETCURRENTKEY("Posting Date");
          GLEntries.SETASCENDING("Posting Date",TRUE);
          IF GLEntries.FINDFIRST THEN BEGIN
            DateFrom := FORMAT((GLEntries."Posting Date"),0,'<Month Text> <Closing><Day>, <Year4>');
          END;
          DateTo := FORMAT(("G/L Entry".GETRANGEMAX("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          IF STRLEN(DateTo) > 0 THEN
            DateTo := ' to ' + DateTo;
        END
        ELSE IF (STRPOS("G/L Entry".GETFILTER("Posting Date"), '..') = 9) AND (STRLEN("G/L Entry".GETFILTER("Posting Date")) > 10) THEN BEGIN
          //MESSAGE('910');
          DateFrom := FORMAT(("G/L Entry".GETRANGEMIN("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          DateTo := FORMAT(("G/L Entry".GETRANGEMAX("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          IF STRLEN(DateTo) > 0 THEN
            DateTo := ' to ' + DateTo;
        END
        ELSE IF STRPOS("G/L Entry".GETFILTER("Posting Date"), '..') = 9 THEN BEGIN
          //MESSAGE('9');
          DateFrom := FORMAT(("G/L Entry".GETRANGEMIN("Posting Date")),0,'<Month Text> <Closing><Day>, <Year4>');
          GLEntries.RESET;
          GLEntries.SETCURRENTKEY("Posting Date");
          GLEntries.SETASCENDING("Posting Date",TRUE);
          IF GLEntries.FINDLAST THEN BEGIN
            DateTo := FORMAT((GLEntries."Posting Date"),0,'<Month Text> <Closing><Day>, <Year4>');
            IF STRLEN(DateTo) > 0 THEN
              DateTo := ' to ' + DateTo;
          END;
        END;
      END
      ELSE BEGIN
        GLEntries.RESET;
        GLEntries.SETCURRENTKEY("Posting Date");
        GLEntries.SETASCENDING("Posting Date",TRUE);
        IF GLEntries.FINDFIRST THEN
          DateFrom := FORMAT(GLEntries."Posting Date");
        IF GLEntries.FINDLAST THEN
          DateTo := FORMAT(GLEntries."Posting Date");
        IF STRLEN(DateTo) > 0 THEN
          DateTo := ' to ' + DateTo;
      END;
    END;

    BEGIN
    END.
