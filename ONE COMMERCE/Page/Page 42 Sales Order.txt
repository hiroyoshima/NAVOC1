    LOCAL PROCEDURE GenerateWhiteLines@51();
    VAR
      SalesLineInit@1003 : Record 37;
      SalesLineToIns@1002 : Integer;
      ControlNo@1001 : Integer;
      SalesLineLastNo@1000 : Integer;
    BEGIN
      SalesLine.SETRANGE("Document Type", SalesLine."Document Type"::Order);
      SalesLine.SETFILTER("Document No.",Rec."No.");

      IF SalesLine.COUNT > 12 THEN
        ERROR(BSReportError);

      SalesLineToIns := 12 - SalesLine.COUNT;
       IF SalesLineToIns > 1 THEN BEGIN
         // SalesLineToIns += 1;
         FOR ControlNo := 1 TO SalesLineToIns DO BEGIN
             SalesLine.FINDLAST;
             SalesLineInit.INIT;
             SalesLineInit."Line No." := (SalesLine."Line No." + 10000);
             SalesLineInit."Document Type" := SalesLineInit."Document Type"::Order;
             SalesLineInit.Type := SalesLineInit.Type::" ";
             SalesLineInit."Document No." := "No.";
             SalesLineInit.Description := '          ';
             SalesLineInit.VALIDATE("Prepmt. VAT Calc. Type");
             SalesLineInit.INSERT(TRUE);
         END;
       END;
    END;

    BEGIN
    END.
