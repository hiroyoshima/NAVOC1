                      Name=Request for Purchase;
                      Image=TestReport;
                      OnAction=VAR
                                 PurchLine@1000 : Record 39;
                                 PurchHeader@1002 : Record 38;
                                 NoRFPurLines@1001 : TextConst 'ENU=No %1 from %2.';
                                 PurchHeaderTmp@1004 : TEMPORARY Record 38;
                                 RFPurNo@1003 : Text;
                               BEGIN
                                 PurchHeaderTmp.DELETEALL;
                                 RFPurNo := '';
                                 PurchLine.RESET;
                                 PurchLine.SETRANGE("Document No.", Rec."No.");
                                 PurchLine.SETRANGE("Document Type", PurchLine."Document Type"::Order);
                                 IF PurchLine.FINDSET THEN BEGIN
                                   REPEAT
                                     IF (PurchHeader.GET(PurchHeader."Document Type"::Invoice, PurchLine."RFP Document No.")) AND (NOT PurchHeaderTmp.GET(PurchHeader."Document Type"::Invoice, PurchLine."RFP Document No.")) THEN BEGIN
                                         PurchHeaderTmp := PurchHeader;
                                       PurchHeaderTmp.INSERT(TRUE);
                                       COMMIT;
                                       IF RFPurNo <> '' THEN
                                         RFPurNo += '|' + PurchHeader."No."
                                       ELSE
                                         RFPurNo := PurchHeader."No.";
                                     END;
                                   UNTIL PurchLine.NEXT = 0;
                                 END;

                                 IF RFPurNo <> '' THEN BEGIN
                                   PurchHeader.RESET;
                                   PurchHeader.SETFILTER("No.", RFPurNo);
                                   PurchHeader.SETRANGE("Document Type", PurchaseHeader."Document Type"::Invoice);
                                   REPORT.RUN(REPORT::"Request for Purchase", TRUE, TRUE, PurchHeader);
                                 END
                                 ELSE
                                   MESSAGE(NoRFPurLines, PurchaseLines.TABLECAPTION, PurchaseHeader."Request Type"::"Request for Purchase");
                               END;
                                }
