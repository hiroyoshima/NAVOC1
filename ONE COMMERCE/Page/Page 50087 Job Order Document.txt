    LOCAL PROCEDURE DocNoVisibility@1();
    VAR
      NoSeries@1000 : Record 308;
    BEGIN
      CustomSetup.GET;
      IF CustomSetup."Job Order Nos." <> '' THEN BEGIN
        IF NoSeries.GET(CustomSetup."Job Order Nos.") AND (NoSeries."Manual Nos." = FALSE) THEN BEGIN
          DocNoVisible := FALSE;
        END ELSE BEGIN
          DocNoVisible := TRUE;
        END;
      END;
    END;
