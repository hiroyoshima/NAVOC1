    LOCAL PROCEDURE GetBlobField@8();
    BEGIN

      CLEAR(DescrptionOfRequestTxt);
      CALCFIELDS("Description of Req. Background");
      IF "Description of Req. Background".HASVALUE THEN BEGIN
        "Description of Req. Background".CREATEINSTREAM(InStr);
        InStr.READ(DescrptionOfRequestTxt);
      END;

      CLEAR(TransactionBackgroundTxt);
      CALCFIELDS("Justif. Assessment of Risk");
      IF "Justif. Assessment of Risk".HASVALUE THEN BEGIN
        "Justif. Assessment of Risk".CREATEINSTREAM(InStr);
        InStr.READ(TransactionBackgroundTxt);
      END;

      CLEAR(CMSRemarksTxt);
      CALCFIELDS(Remarks);
      IF Remarks.HASVALUE THEN BEGIN
      Remarks.CREATEINSTREAM(InStr);
      InStr.READ(CMSRemarksTxt);
      END;
    END;
