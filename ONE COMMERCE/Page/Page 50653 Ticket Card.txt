    LOCAL PROCEDURE SetControlVisibility@7();
    VAR
      ApprovalsMgmt@1002 : Codeunit 1535;
    BEGIN
      OpenApprovalEntriesExistForCurrUser := ApprovalsMgmt.HasOpenApprovalEntriesForCurrentUser(RECORDID);
      OpenApprovalEntriesExist := ApprovalsMgmt.HasOpenApprovalEntries(RECORDID);
      HasIncomingDocument := "Incoming Document Entry No." <> 0;
      IF "Error Code" <> '' THEN
        IsEditable := FALSE;

      IsAdditionalInfoEditable := CurrPage.EDITABLE;
    END;

    LOCAL PROCEDURE GetBlobField@1();
    BEGIN
      CALCFIELDS(Recipients);
      CLEAR(RecipientsText);
      IF Recipients.HASVALUE THEN BEGIN
        Recipients.CREATEINSTREAM(InStr);
        InStr.READ(RecipientsText);
      END;

      CALCFIELDS("Additional Information");
      CLEAR(AdditionalInfoText);
      IF "Additional Information".HASVALUE THEN BEGIN
        "Additional Information".CREATEINSTREAM(InStr);
        InStr.READ(AdditionalInfoText);
      END;


      CALCFIELDS("Error Text");
      CLEAR(ErrorTextText);
      IF "Error Text".HASVALUE THEN BEGIN
       "Error Text".CREATEINSTREAM(InStr);
       InStr.READ(ErrorTextText);
      END;

    END;

    BEGIN
    END.
