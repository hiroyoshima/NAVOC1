                                BEGIN
                                  VendCustName := '';
                                  UserGroupCode := '';
                                  UserGroup := '';
                                  ProjectName := '';
                                  CLEAR(AppliedDocNo);
                                  CLEAR(AppliedExternalDocNo);
                                  CLEAR(INTPNo);
                                  CASE "Bank Account Ledger Entry"."Bal. Account Type" OF
                                    "Bank Account Ledger Entry"."Bal. Account Type"::Customer:
                                      BEGIN
                                        Customer.GET("Bank Account Ledger Entry"."Bal. Account No.");
                                        VendCustName := Customer.Name + ' ' + Customer."Name 2";
                                      END;
                                    "Bank Account Ledger Entry"."Bal. Account Type"::Vendor:
                                      BEGIN
                                        Vendor.GET("Bank Account Ledger Entry"."Bal. Account No.");
                                        VendCustName := Vendor.Name + ' ' + Vendor."Name 2";
                                        IF Vendor."Vendor Type" = Vendor."Vendor Type"::Employee THEN
                                          IF Employee.GET(Vendor."No.") THEN BEGIN
                                            Employee.CALCFIELDS(Department);
                                            UserGroupCode := Employee."Department Code";
                                            UserGroup := Employee.Department;
                                          END;
                                        VendorLedgerEntries.SETRANGE("Transaction No.", "Bank Account Ledger Entry"."Transaction No.");
                                        IF VendorLedgerEntries.FINDFIRST THEN BEGIN
                                          AppliedExternalDocNo := VendorLedgerEntries."Applies-to Ext. Doc. No.";
                                          IF VendorLedgerEntries."Applies-to Doc. No." <> '' THEN
                                            AppliedDocNo := VendorLedgerEntries."Applies-to Doc. No."
                                          ELSE
                                            AppliedDocNo := VendorLedgerEntries."Applies-to Doc. No. 2";
                                        END;
                                      END;
                                  END;

                                  IF "Bank Account Ledger Entry"."Global Dimension 1 Code" <> '' THEN BEGIN
                                    DimensionValue.SETFILTER("Dimension Code", 'PROJECT');
                                    DimensionValue.SETRANGE(Code, "Bank Account Ledger Entry"."Global Dimension 1 Code");
                                    IF DimensionValue.FINDFIRST THEN
                                      ProjectName := DimensionValue.Name;

                                    DefaultDimension.SETRANGE("Table ID", DATABASE::"Internal Notice to Proceed");
                                    DefaultDimension.SETRANGE("Dimension Code", 'PROJECT');
                                    DefaultDimension.SETRANGE("Dimension Value Code", "Bank Account Ledger Entry"."Global Dimension 1 Code");
                                    IF DefaultDimension.FINDFIRST THEN
                                      INTPNo := DefaultDimension."No.";
                                  END;
                                END;
                                 }
