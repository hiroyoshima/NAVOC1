    PROCEDURE WriteCSV@3(LocString@1000 : ARRAY [10] OF Text[250]);
    VAR
      ArrayCtr@1001 : Integer;
      TextToWrite@1002 : Text[250];
    BEGIN
      FOR ArrayCtr := 1 TO 10 DO BEGIN
        TextToWrite += ApplyQuote(FORMAT(LocString[ArrayCtr])) + ','
      END;
      TextToWrite := DELSTR(TextToWrite,STRLEN(TextToWrite),1);
      BIRFile.WRITE(TextToWrite);
      CLEAR(LocString)
    END;

    PROCEDURE ApplyQuote@5(LocString@1000 : Text[250]) : Text[250];
    VAR
      CharCtr@1002 : Integer;
      ApplyQuote@1001 : Boolean;
    BEGIN
      ApplyQuote := FALSE;
      FOR CharCtr := 1 TO STRLEN(LocString) DO BEGIN
        IF LocString[CharCtr] = ',' THEN BEGIN
          EXIT('"' + LocString + '"');
        END;
      END;

      EXIT (LocString);
    END;

    BEGIN
