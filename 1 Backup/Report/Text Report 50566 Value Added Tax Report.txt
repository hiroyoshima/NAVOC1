    LOCAL PROCEDURE ReplaceHashWithRegEx@120(InputStr@1000 : Text) OutputStr : Text;
    VAR
      Position@1001 : Integer;
    BEGIN
      Position := STRPOS(InputStr,'#');
      IF Position = 0 THEN
        EXIT(InputStr);
      OutputStr := DELSTR(InputStr,Position,1);
      OutputStr := INSSTR(OutputStr,'%23',Position);
    END;

    BEGIN
    END.
