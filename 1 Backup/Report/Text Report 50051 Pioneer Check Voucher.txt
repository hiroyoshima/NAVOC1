    LOCAL PROCEDURE GetPostingAccount@2(AcccountType@1001 : 'G/L Account,Customer,Vendor,Bank Account';AccountCode@1000 : Code[20]) : Code[20];
    VAR
      GLAccount@1002 : Record 15;
      Vendor@1003 : Record 23;
      Customer@1004 : Record 18;
      Bank@1005 : Record 270;
      CustomerPostingGroup@1007 : Record 92;
      VendorPostingGroup@1006 : Record 93;
      BankAccountPostingGroup@1008 : Record 277;
    BEGIN
      CASE AcccountType OF
        AcccountType::"G/L Account":
          EXIT(AccountCode);
        AcccountType::Customer: BEGIN
          IF Customer.GET(AccountCode) THEN
            IF CustomerPostingGroup.GET(Customer."Customer Posting Group") THEN
              EXIT(CustomerPostingGroup."Receivables Account");
        END;
        AcccountType::Vendor: BEGIN
        IF Vendor.GET(AccountCode) THEN
          IF VendorPostingGroup.GET(Vendor."Vendor Posting Group") THEN
            EXIT(VendorPostingGroup."Payables Account");
        END;
        AcccountType::"Bank Account":BEGIN
          IF Bank.GET(AccountCode) THEN
            IF BankAccountPostingGroup.GET(Bank."Bank Acc. Posting Group") THEN
              EXIT(BankAccountPostingGroup."G/L Bank Account No.");
        END;
      END;
    END;

    LOCAL PROCEDURE InsertPaymentEntry@5(AccountClassification@1000 : 'Account Type,Balancing Account Type,Unerealized VAT,Realized VAT,WHT Tax');
    VAR
      GLAccount@1003 : Record 15;
      VatPostingSetup@1004 : Record 325;
      WHTPostingSetup@1005 : Record 50503;
      GLAccountCode@1001 : Code[20];
      GLAccountName@1002 : Text;
    BEGIN
      CASE AccountClassification OF
        AccountClassification::"Balancing Account Type":
          GLAccountCode := GetPostingAccount(GenJrnlLine."Bal. Account Type",GenJrnlLine."Bal. Account No.");
        AccountClassification::"Account Type":
        GLAccountCode := GetPostingAccount(GenJrnlLine."Account Type",GenJrnlLine."Account No.");
        AccountClassification::"Unerealized VAT":
          BEGIN
            VatPostingSetup.GET(GenJrnlLine."VAT Bus. Posting Group", GenJrnlLine."VAT Prod. Posting Group");
            IF VatPostingSetup."Unrealized VAT Type" = VatPostingSetup."Unrealized VAT Type"::" " THEN
              VatPostingSetup.FIELDERROR("Unrealized VAT Type");
            VatPostingSetup.TESTFIELD("Purch. VAT Unreal. Account");
            GLAccountCode := VatPostingSetup."Purch. VAT Unreal. Account";
          END;
        AccountClassification::"WHT Tax":
          BEGIN
            IF NOT WHTPostingSetup.GET(GenJrnlLine."WHT Business Posting Group", GenJrnlLine."WHT Product Posting Group") OR
              (WHTPostingSetup."Realized WHT Type" <> WHTPostingSetup."Realized WHT Type"::Payment) THEN
              EXIT;
            WHTPostingSetup.TESTFIELD("Payable WHT Account Code");
            GLAccountCode := WHTPostingSetup."Payable WHT Account Code";
          END;
        ELSE
          EXIT;
      END;


      IF GLAccount.GET(GLAccountCode) THEN
        GLAccountName := GLAccount.Name;

      EntryNo += 1;

      PaymentGLEntry.INIT;
      PaymentGLEntry."Entry No." := EntryNo;
      PaymentGLEntry."G/L Account No." := GLAccountCode;
      PaymentGLEntry."G/L Account Name" := GLAccountName;
      PaymentGLEntry."Document No." := GenJrnlLine."Document No.";

      CASE AccountClassification OF
        AccountClassification::"Balancing Account Type", AccountClassification::"Unerealized VAT":
          BEGIN
            PaymentGLEntry.Amount := GenJrnlLine.Amount * -1;
            PaymentGLEntry."Credit Amount" := GenJrnlLine.Amount;
            PaymentGLEntry."Debit Amount" := 0;
          END;
        AccountClassification::"WHT Tax":
          BEGIN
            PaymentGLEntry.Amount := CalcWHT(GenJrnlLine);
            PaymentGLEntry."Credit Amount" := CalcWHT(GenJrnlLine);
            PaymentGLEntry."Debit Amount" := 0;
          END;
        AccountClassification::"Account Type":
          BEGIN
            PaymentGLEntry.Amount := GenJrnlLine.Amount;
            PaymentGLEntry."Debit Amount" := GenJrnlLine.Amount;
            PaymentGLEntry."Credit Amount" := 0;
          END;
      END;
      PaymentGLEntry."Global Dimension 1 Code" := GenJrnlLine."Shortcut Dimension 1 Code";
      PaymentGLEntry."Global Dimension 2 Code" := GenJrnlLine."Shortcut Dimension 2 Code";
      PaymentGLEntry."Dimension Set ID" := GenJrnlLine."Dimension Set ID";
      PaymentGLEntry.INSERT;
    END;

    LOCAL PROCEDURE CalcWHT@18(VAR GenJnlLine@1000 : Record 81) : Decimal;
    VAR
      WHTSetup@1001 : Record 50503;
      GenJnlLine1@1002 : Record 81;
      WHTManagement@1003 : Codeunit 50500;
      WHTAmount@1004 : Decimal;
    BEGIN
      GenJnlLine1.COPY(GenJnlLine);
      WHTAmount := 0;
      IF GenJnlLine1."Interest Amount" <> 0 THEN
        GenJnlLine1.VALIDATE(Amount,GenJnlLine1.Amount - GenJnlLine1."Interest Amount");
      IF GenJnlLine."Account Type" = GenJnlLine."Account Type"::Vendor THEN
        IF WHTSetup.GET(GenJnlLine."WHT Business Posting Group",GenJnlLine."WHT Product Posting Group") THEN
          IF WHTSetup."Realized WHT Type" <> WHTSetup."Realized WHT Type"::Earliest THEN
            WHTAmount := WHTAmount + WHTManagement.WHTAmountJournal(GenJnlLine1,FALSE)
          ELSE
            WHTAmount := WHTAmount + ABS(WHTManagement.CalcVendExtraWHTForEarliest(GenJnlLine1));

      IF WHTAmount < 0 THEN
        WHTAmount := 0;

      EXIT(WHTAmount);
    END;
