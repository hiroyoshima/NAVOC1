    PROCEDURE SetStyle@6() : Text;
    BEGIN
      IF NOT (Status = Status::Completed) THEN BEGIN
        IF Status = Status::"Not Started" THEN
          EXIT('');
        IF TODAY > "Finishing Date" THEN
          EXIT('Unfavorable')
      END ELSE
        IF "Date Completed" > "Finishing Date" THEN
          EXIT('Attention');
      EXIT('');
    END;

    LOCAL PROCEDURE SetStatusStyle@21() : Text;
    BEGIN
      CASE Status OF
        Status::Completed:
          EXIT('Favorable');
        Status::"In Progress":
          EXIT('StrongAccent');
        Status::Postponed, Status::"Not Started":
          EXIT('Subordinate');
        ELSE
          EXIT('');
      END;
    END;
