    LOCAL PROCEDURE InitTempTable@27();
    BEGIN
      SalesLine.RESET;
      SalesLine.SETFILTER("Document Type", FORMAT("Document Type"::Invoice));
      SalesLine.SETFILTER("Request type", FORMAT("Request type"::Withdrawal));
      IF SalesLine.FINDSET THEN BEGIN
        REPEAT
          IF SalesHeader.GET(SalesLine."Document Type", SalesLine."Document No.") AND (SalesHeader.Status = SalesHeader.Status::Released) THEN BEGIN
            Rec := SalesLine;
            INSERT;
          END;
        UNTIL SalesLine.NEXT = 0;
      END;
    END;
